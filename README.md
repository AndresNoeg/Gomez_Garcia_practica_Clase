numero=int(input("Ingresa un número: "))

    print("El número es par.") #Se imprimira el sig. Mensaje

else: #Si no se cumplen las 2 condiciones anteriores

    print("El número es impar.") # Se imprimira el sig. Mensaje
    
    print("  ")
